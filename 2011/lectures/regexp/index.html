Regex Example
  • input: http://www.databasebasketball.com/players/playerlist.htm
  • regex:
    var a = input.match(/<td><a href="playerpage\.htm\?ilkid=\w+ ">[^<]+<\/a><br>/g)
for (var i = 0; i < a.length; i++) {
    var b = a[i]
    b = b.match(/<td><a href="playerpage\.htm\?ilkid=\w+ ">([^<]+)<\/a><br>/)
    b = b[1]
    b = b.replace(/&nbsp;/, " ")
    b = b.match(/([^,]+), (.*)/)
    print(b[2] + " " + b[1])
}
